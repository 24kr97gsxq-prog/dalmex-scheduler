-- V3.1.1 FIX: Scope dock uniqueness to non-DMX carriers only -- The recycling dock can only handle 1 outbound + 1 inbound at a time. -- DMX routes happen at customer locations, so they don't consume dock capacity. DROP INDEX IF EXISTS public.idx_schedule_unique_active_slot; -- New rule: among NON-DMX bookings, uniqueness is (time, direction). -- DMX bookings (swaps, round trips, deliveries) are excluded from the rule. CREATE UNIQUE INDEX idx_schedule_unique_active_slot ON public.schedule (time, direction) WHERE status NOT IN ('cancelled', 'no_show') AND carrier NOT ILIKE '%dalmex%'; -- Verify SELECT indexdef FROM pg_indexes WHERE schemaname = 'public' AND tablename = 'schedule' AND indexname = 'idx_schedule_unique_active_slot';
